# Установка ArchLinux

ping 8.8.8.8
iwctl
station wlan0 connect SSID

date
timedatectl set-timezone Asia/Novosibirsk

lsblk
cfdisk -z /dev/nvme0n1

/dev/nvme0n1p1 -- 600M -- EFI 
/dev/nvme0n1p2 -- 40G 	-- /
/dev/nvme0n1p3 -- 425G -- /home
/dev/nvme0n1p4 -- 8GB 	-- Swap

mkfs.fat -F 32 /dev/sda1
mkfs.btrfs /dev/sda2 -L "Arch"
mkfs.btrfs /dev/sda3 -L "Home"

mount -o noatime,ssd,space_cache=v2,compress-force=zstd:3,discard=async /dev/vda2 /mnt
mount --mkdir -o noatime,ssd,space_cache=v2,compress-force=zstd:3,discard=async /dev/vda3 /mnt/home
mount --mkdir /dev/vda1 /mnt/boot

pacstrap -K /mnt base base-devel linux linux-firmware git amd-ucode sudo nano neovim networkmanager iwd dhcpcd tlp openssh zsh
genfstab -U /mnt >> /mnt/etc/Fstab

arch-chroot /mnt

ln -sf /usr/share/zoneinfo/Asia/Novosibirsk /etc/localtime
hwclock --systohc

nano /etc/locale.gen :
  en_US.UTF-8
  ru_RU.UTF-8

locale-gen

echo "LANG=ru_RU.UTF-8" >> /etc/locale.conf

echo -e "KEYMAP=ru \nFONT=cyr-sun16" >> /etc/vconsole.conf

echo "ZenBook" >> /etc/hostname

nano /etc/hosts :
127.0.0.1	localhost
::1         localhost
127.0.1.1	ZenBook

useradd -G wheel -s /bin/bash -m NAME

passwd NAME >> PASSWORD
passwd >> PASSWORD

systemctl enable NetworkManager.service
systemctl enable fstrim.timer
systemctl enable tlp.service
tlp-stat -b

nano /etc/pacman.conf :
  [multilib]
  Include = /etc/pacman.d/mirrorlist

nano /etc/sudoers :
  %wheel      ALL=(ALL:ALL) ALL

bootctl install
bootctl update

nano /boot/loader/loader.conf :
default arch.conf
timeout 
console-mode max
editor yes

blkid /dev/nvme0n1p2 -- копируем PARTUUID корневого раздела

nano /boot/loader/entries/arch.conf :
title	ArchLinux
linux	/vmlinuz-linux
initrd	/amd-ucode.img
initrd	/initramfs-linux.img
options	root=PARTUUID=0e0d7676-3a9e-d74d-ab6d-f7d3abadd63d rw quiet splash loglevel=3 udev.log_level=3 i8042.probe_defer

nano /etc/mkinitcpio.conf:
BINARIES=(setfont)

mkinitcpio -P

exit

reboot
